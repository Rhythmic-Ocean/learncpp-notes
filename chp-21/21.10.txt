-Overriding operator(), also called functors
  -only operator that can take variable arguments (unlike other operator with fixed argument)
 NOTE: starting C++23 [] operator can also take variable arguments!! 
  eg:
  -can ONLY BE IMPLEMENTED AS MEMBER FUNCTION!!

      class Matrix
      {
      private:
          double m_data[4][4]{};

      public:
          double& operator()(int row, int col);
          double operator()(int row, int col) const; // for const objects
      };

      double& Matrix::operator()(int row, int col)
      {
          assert(row >= 0 && row < 4);
          assert(col >= 0 && col < 4);

          return m_data[row][col];
      }

      double Matrix::operator()(int row, int col) const
      {
          assert(row >= 0 && row < 4);
          assert(col >= 0 && col < 4);

          return m_data[row][col];
      }

      int main()
      {
          Matrix matrix;
          matrix(1, 2) = 4.5;
          std::cout << matrix(1, 2) << '\n';

          return 0;
      }
