  I/O overloading:
    -when we do std::cout<<5<<"me";, operator << has been overloaded for std::ostream (which is like patent type of std::cout) and int/ string to print and return std::cout back
        So, std::cout<<5, prints 5 to console and returns std::cout and std::cout<<"me" will print "me" and return std::cout again!
    -we can do something similar as above and overload std::cout and our custom enum type for operator<<
      eg:
      enum Color{
        none, 
        red,
        blue,
        green,
      }

      constexpr std::string_view StringFromEnum (Color c){
        switch(c){
          case red:
            return "red";
          case blue:
            return "blue";
          case green:
            return "green";
          default:
            return "???";
        }
      }

      constexpr std::optional<Color> EnumFromString (std::string_view ssv){
        if(ssv == "red") return red;
        if(ssv == "blue") return blue;
        if(ssv=="green") return green;
        return {};
      }

      std::ostream& operator<<(std::ostream& out, Color color){
        out << StringFromEnum(color);
        return out;
      }

      std::istream& operator>>(std::istream& in, Color& color){
        std::string str{};
        in >> str;
        std::optional<Color> match {EnumFromString(color)};
        if(match){
          color = *match;
          return in;
        }
        in.setstate(std::ios_base::failbit);
        return in;
      }

      int main(){
        Color color {red};
        std::cout << red;
        Color colour {};
        std::cin >> color;
        if(std::cin) std::cout << "You choose" <<  color;
        else{
            std::cin.clear();
            std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
            std::cout << "Invalid input";
          }
        return 0;
      }

13.9
  -at least not zero initializing members of struct when defining them is bad!
  -explicit values used during struct intiialization takes precedence over default ones used
      struct Something
      {
          int x;       // no default initialization value (bad)
          int y {};    // value-initialized by default
          int z { 2 }; // explicit default value
      };

      int main()
      {
          Something s1;             // No initializer list: s1.x is uninitialized, s1.y and s1.z use defaults
          Something s2 { 5, 6, 7 }; // Explicit initializers: s2.x, s2.y, and s2.z use explicit values (no default values are used)
          Something s3 {};          // Missing initializers: s3.x is value initialized, s3.y and s3.z use defaults

          return 0;
      }
  -Provide a default value for all members. This ensures that your members will be initialized even if the variable definition doesn’t include an initializer list.
    Fraction f1;          
    Fraction f2 {}; -> better

  C++17: Class Template Argument Deduction (CATD):
    #include <utility>
    int main(){
      std::pair<int, int> p  {1, 2}; 
      std::pair p{1, 2}; -> legal! Compiler automatically deduces the type
    }
    
    BUT
    int main(){
      Pair<int, int> p {1, 2}; 
      Pair pp{1, 2}; -> illegal in C++17, legal in c++20 and above
    } // CATD can't deduce template type for aggregrate types in c++ 17 for that we need template argument deduction guide

    template <typename T, typename U>
    struct Pair{
      T first{};
      T sec{};
    }

    template <typename T, typename U>
    Pair(T, U) -> Pair<T, U>

    int main(){
      Pair<int, int> p {1, 2}; 
      Pair pp{1, 2}; -> legal in C++17
    } 

  ANother example:
    template <typename T>
    struct Pair
    {
        T first{};
        T second{};
    };

    // Here's a deduction guide for our Pair (needed in C++17 only)
    // Pair objects initialized with arguments of type T and T should deduce to Pair<T>
    template <typename T>
    Pair(T, T) -> Pair<T>;

  NOTE: Non-aggregates don’t need deduction guides in C++17 because the presence of a constructor serves the same purpose!!
  NOTE (for non type template argument from prev chap) template<int N> void foo() generates a unique function for every distinct value of N!!

    template <typename T = int, typename U = double>

    ooo(T, U) -> ooo<T, U>

     }
